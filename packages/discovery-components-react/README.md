## Install

```bash
npm install --save @disco-widgets/react-components
```

## Usage

```tsx
import * as React from 'react';
import MyComponent from '@disco-widgets/react-components';

class Example extends React.Component<Props, object> {
  render() {
    return <MyComponent />;
  }
}
```
